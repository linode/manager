import { EditableText } from './EditableText';
import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs';
import { useArgs } from '@storybook/client-api';

<Meta title="Components/Editable Text" component={EditableText} />

# Editable Text

export const Template = (args, context) => {
  const [localArgs, setLocalArgs] = useArgs();
  const onEdit = (updatedText) => {
    return Promise.resolve(setLocalArgs({text: updatedText}));
  }
  return <EditableText text={localArgs.text} onEdit={onEdit} onCancel={() => {}}/>
}

<Canvas>
  <Story 
    name="Headline & Title"
    args={{
      text: 'Edit Me!',
    }}>
    {Template.bind({})}
  </Story>
</Canvas>

<ArgsTable story="Headline & Title" include={['onEdit', 'onCancel', 'text']} sort="requiredFirst"/>
