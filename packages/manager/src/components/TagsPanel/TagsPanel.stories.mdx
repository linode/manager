# Tags Panel

Currently used in:

- `DomainRecordsWrapper`
- `KubeSummaryPanel`
- Linode and NodeBalancer - `SummaryPanel`

export const Template = (args) => {
  const [tags, setTags] = React.useState([]);
  const updateTags = (tags) => {
    return new Promise((resolve) => {
      setTimeout(() => {
        setTags(tags);
        resolve();
      }, 500);
    });
  };
  return <TagsPanel {...args} tags={tags} updateTags={updateTags} />;
};
