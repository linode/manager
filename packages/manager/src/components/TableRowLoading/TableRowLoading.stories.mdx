# Table Row Loading

export const Template = (args) => {
  const heads = [];
  const unit = 100 / args.columns;
  for (let i = 0; i < args.columns; i++) {
    heads.push(<TableCell style={{ width: `${unit}%` }}>Label {i}</TableCell>);
  }
  return (
    <Table>
      <TableHead>
        <TableRow>
          {heads}
        </TableRow>
      </TableHead>
      <TableBody>
        <TableRowLoading {...args} />
      </TableBody>
    </Table>
  );
};
