# Action Menu

### Usage

No more than 8 items should be displayed within an action menu.

export const actions = [
  {
    title: 'First Action',
    onClick: () => null,
  },
  {
    title: 'Second Action',
    onClick: () => null,
  },
  {
    title: 'Disabled Action',
    disabled: true,
    onClick: () => null,
    tooltip: 'An explanation as to why this item is disabled',
  },
];

## Standard actions

<Canvas>
  <Story name="Action Menu">
    <ActionMenu
      actionsList={actions.filter((action) => !action.disabled)}
      ariaLabel="label"
    />
  </Story>
</Canvas>
