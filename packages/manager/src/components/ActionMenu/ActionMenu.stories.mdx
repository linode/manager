# Action Menu

### Usage

No more than 8 items should be displayed within an action menu.

export const standardActions = [
  {
    title: 'First Action',
  },
  {
    title: 'Second Action',
  },
];

export const disabledActions = {
  title: 'Disabled Action',
  disabled: true,
  tooltip: 'An explanation as to why this item is disabled',
};

export const Template = (args) => {
  const { actionsList } = args;
  const cleanedActionsList = actionsList.map(({onClick, title, ...rest}) => ({ onClick: action(`${title} clicked`), title: title, ...rest}));
  return (
    <ActionMenu ariaLabel="label" actionsList={cleanedActionsList} />
  );
}
