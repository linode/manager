# ActionsPanel

### Usage

An ActionsPanel presents a primary and secondary action to the user, though it can also be used to render a single action within modals or drawers for consistency.

export const primaryButtonProps =
  {
    label: 'Confirm',
    onClick: action('Confirm clicked'),
  };

export const secondaryButtonProps =
  {
    label: 'Cancel',
    onClick: action('Cancel clicked'),
  };

export const disabledPrimaryButtonProps = {
  ...primaryButtonProps, 
  onClick: action('Disabled Action clicked'),
  disabled: true,
};

export const disabledSecondaryButtonProps = {
  ...secondaryButtonProps, 
  onClick: action('Disabled Action clicked'),
  disabled: true,
};

export const Template = (args) => {
  return <ActionsPanel {...args} />
}

### Standard actions

export const standardActions = {
  primaryButtonProps: primaryButtonProps,
  secondaryButtonProps: secondaryButtonProps
};

export const standardAndDisabledActions = {
  primaryButtonProps: disabledPrimaryButtonProps,
  secondaryButtonProps: disabledSecondaryButtonProps
};
