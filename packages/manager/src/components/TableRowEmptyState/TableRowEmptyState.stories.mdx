# Table Row Empty State

- Tables that do not yet have any data should be shown full width with the title and column heads. This letâ€™s users know where data will appear when it is available.
- An empty table should have language that clearly says why the table has no data and what action to take to populate with data.

export const Template = (args) => {
  const fields = [];
  for (let i = 0; i < args.colSpan; i++) {
    fields.push(
      <TableCell>
        Label {i}
      </TableCell>
    );
  }
  return (
    <Table>
      <TableHead>
        <TableRow>
          {fields}
        </TableRow>
      </TableHead>
      <TableBody>
        <TableRowEmptyState {...args} />
      </TableBody>
    </Table>
  );
}
